<mx:SWFLoader 
    xmlns:mx="http://www.adobe.com/2006/mxml" implements="smalldebugger.IListenerOnDebug"
    source="@Embed('/assets/roundbag.swf')"
    trustContent="true"
    >
    <mx:Script>
        <![CDATA[
            /**
             Визуальный компонент представляющий собой вертящуюся штуковину в верху окна программы, каждый раз когда приходит debug сообщение
             в основе своей использован flash9 ролик с вызываемой функцией DoRotate запускающей процесс вращения
            */
            import mx.managers.SystemManager;
            public var _listener : SmallDebuggerPool = null;
            
            [Bindable]            
            public function get listener ():SmallDebuggerPool{
                return _listener;
            }
            public function set listener (_nlist : SmallDebuggerPool):void{
                 if (_listener!=null)
                     _listener.removeOnDebugListener (this);
                 _listener = _nlist;
                 _listener.addOnDebugListener (this);
            }
            
            
            /**
             Реализация интерфейса IListenerOnDebug
            */
            public function onDebug (record : SmallDebugRecord):void{
                var mc : MovieClip = MovieClip(content);
                var loa : Loader = mc.getChildAt(0) as Loader;
                var DoRotate :Function = loa.content['DoRotate'] as Function;
                DoRotate.call();
            }
            
        ]]>
    </mx:Script>
</mx:SWFLoader>
