<?xml version="1.0" encoding="utf-8"?>
<mx:TitleWindow 
    xmlns:mx="http://www.adobe.com/2006/mxml" 
    layout="absolute" 
    width="330" height="102"
    implements="logic.IPopupWhileLoadingControl"
    title="Ждите, идет загрузка данных с сервера"
    >
    <mx:Script>
        <![CDATA[
        import mx.core.Application;
        import mx.core.IFlexDisplayObject;
        import mx.managers.PopUpManager;
        
        /**
        * Счетчик количество показов данного окна, когда это число положительное то окно видимо
        * как только число становится равным нулую то окно прячется
        * одно окно может быть использовано для любого количества HessianDataSet-ов одновременно
        */
        [Bindable]
        protected var counter_shows : int = 0;
        public function onStart ():void{
            counter_shows++;
            if (counter_shows == 1){
                PopUpManager.addPopUp(this, DisplayObject(Application.application), true);
                PopUpManager.centerPopUp(this);
            }
        }
        public function onStop ():void{
            counter_shows--;            
            if (counter_shows == 0){
                PopUpManager.removePopUp(this);
            }
        }
        public function getSelf ():IFlexDisplayObject{
            return this;
        }
            
        ]]>
    </mx:Script>    
    <mx:Label text="Количество запрошенных ресурсов: {counter_shows}" />
    <mx:ProgressBar x="55" y="26" id="pbar" indeterminate="true"/>
</mx:TitleWindow>
