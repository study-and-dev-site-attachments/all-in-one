package logic
{
    import flash.events.Event;

    /**
     * Класс событие служит для извещения заинтересованных слушателей об том что набор данных меняет внутренее состояние
     * За конкретный тип смены состояния отвечает значение stateKind
     */ 
    public class HessianScrollEvent extends Event
    {

        /** 
         * обработана команда перехода в режим редактирования
         */ 
        public static var DATASET_TRIGGERED_EDIT : String = "edit";        
        /** 
         * обработана команда перехода в добавления новой записи
         */ 
        public static var DATASET_TRIGGERED_APPEND : String = "append";
        /** 
         * обработана команда перехода в режим сохранения изменений
         */ 
        public static var DATASET_TRIGGERED_POST : String = "post";
        /** 
         * обработана команда перехода в режим удаления записи
         */ 
        public static var DATASET_TRIGGERED_DELETE : String = "delete";
        /** 
         * обработана команда перехода в режим отмены редактирования
         */ 
        public static var DATASET_TRIGGERED_CANCEL : String = "cancel";
        /** 
         *  было выполнено произвольное перемещение указателя на текущую запись
         */ 
        public static var DATASET_TRIGGERED_SCROLL : String = "scroll";
        /**
         *  было выполнено редактирование одного из полей
         */ 
        public static var DATASET_TRIGGERED_MODIFY : String = "modify";        
        
        
        
        /** 
         * обработана команда перехода к первой записи
         */ 
        //public static var DATASET_TRIGGERED_FIRST : String = "move_to_first";
        /** 
         * обработана команда перехода к следующей записи
         */ 
        //    public static var DATASET_TRIGGERED_NEXT : String = "move_to_next";
        /** 
         * обработана команда перехода к предыдущей записи
         */ 
        // public static var DATASET_TRIGGERED_PREV : String = "move_to_prev";
        /** 
         * обработана команда перехода к последней записи в наборе
         */ 
        // public static var DATASET_TRIGGERED_LAST : String = "move_to_last";
                
        
        
        public var stateKind : String;
        
        public function HessianScrollEvent (stateKind : String):void{
            super (stateKind);
            this.stateKind  = stateKind;
        }
    }
}
