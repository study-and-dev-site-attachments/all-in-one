package logic
{
    import mx.containers.TitleWindow;
    import user_components.logic.SQLAlertWindow;
    import mx.managers.PopUpManager;
    import mx.core.Application;
    import flash.display.DisplayObject;
    
    public class SQLAlert
    {
        public static function show (msg : String):void{
            var winmsg : SQLAlertWindow = new  SQLAlertWindow();
            winmsg.InitSQLAlertWindow (msg);
            PopUpManager.addPopUp(winmsg, DisplayObject(Application.application), true);
            PopUpManager.centerPopUp(winmsg);
        }
    }
}
