package logic
{
    import mx.controls.Alert;
    
    /**
     * Вспомогательный класс содержит строку адреса подключения к серверу hessian
     */ 
    public class GlobalParams
    {
        protected static const URL_2_HESSIAN_WEB : String  = "http://java.black-zorro.com/";
        protected static const URL_2_HESSIAN_HOME : String  = "http://center:8080/flex/";        

        protected static const SERVICE_ADMIN : String  = "hessian/mysqlobmachadmin";
        protected static const SERVICE_ANALYTICS : String  = "hessian/mysqlobmachanalytics";        
        
        public static function getHessianURL (kind : String):String{
            var base : String = "";
            if (false)
             base = URL_2_HESSIAN_HOME;
            else
             base = URL_2_HESSIAN_WEB;
            
            if (kind == "admin") return base + SERVICE_ADMIN;
            if (kind == "analytics") return base + SERVICE_ANALYTICS;            
            mx.controls.Alert.show("Ошибка запрошено веб-имя неизвестного сервиса: " + kind);
            return null; 
        }
        
        
    }
}
