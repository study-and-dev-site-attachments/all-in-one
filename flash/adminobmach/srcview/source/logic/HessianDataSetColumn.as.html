package logic
{
    /**
     * Класс являющийся частью пользовательского представления загруженных с сервера данных
     * Хранит сведения об отдельной колонке результата
     */ 
    public class HessianDataSetColumn
    {
        /**
         * Имя поля
         */ 
        public var columnName : String;
        /**
         * Имя таблицы откуда были загружены данные
         */ 
        public var tableName : String;
        /**
         * Тип поля
         */ 
        public var columnType : String;
        /**
        * Размер поля
        */ 
        public var columnSize : int;
        /**
         * Признак того является ли данное поле первичным ключом записи
         */ 
        public var isPrimaryKey : Boolean;
        
        /**
         * Признак того является ли данное поле булева типа данных
         */ 
        public var isBoolean : Boolean;
        
        
        
        public function HessianDataSetColumn (column:Object){
            columnName = column["columnName"];
            tableName  = column["tableName"];
            columnType = column["columnType"];
            columnSize = column["columnSize"];
            isPrimaryKey = column["isPrimaryKey"];
            isBoolean = column["isBoolean"];
            
        }

    }
}
